\textbf{Definition:} En tvådimensionell stokastisk variabel är en funktion $(X,Y)$ definierad på ett utfallsrum $\Omega$ och som
tar värden i $\mathbb{R}^2$. 
\begin{itemize}

\item Då man man samtidigt studerar två slumpmässigt varierande storheter $X$ och $Y$ får man en tvådimensionell slumpvariabel.

\begin{frame}
\frametitle{Tvådimensionell kontinuerlig fördelning*}

Låt $(X,Y)$ ha täthetsfunktionen
$$
f_{X,Y}(x,y)=
\begin{cases}
x+y, \quad \text{om } 0<x<1, \text{ } 0<y<1,\\
0, \quad \text{annars}.
\end{cases}
$$
\pause
Den marginella täthetsfunktionen för $Y$ är då
$$
f_Y(y)=\int_{-\infty}^{{\infty}}f_{X,Y}(x,y)dx=\int_{0}^{{1}}(\textcolor{red}{x}+y)\textcolor{red}{dx}=[\textcolor{red}{\frac{x^2}{2}}+y\cdot\textcolor{red}{x}]_{0}^{1}=\frac{1}{2}+y.
$$
\pause
Den betingade täthetsfunktionen för $X$, givet att $Y=y$, blir då
$$
f_{X|Y=y}(x)=\frac{f_{X,Y}(x,y)}{f_Y(y)}=\frac{x+y}{\frac{1}{2}+y}.
$$

\begin{frame}%#14
\frametitle{Linearity of Expectations (Section 7.4.3)}

\begin{theorem}
Let $a$ and $b$ be real numbers and $X_1$ and $X_2$ be a discrete RV.
Then
\begin{enumerate} 
\item$E[X_1+X_2]=E[X_1]+E[X_2]$, and

\item $E[aX+b]=aE[X]+b.$
\end{enumerate}
\end{theorem}


\begin{lemma}
Suppose that $X\sim Bernoulli(p)$. Then $E[X]=p$.
\end{lemma}


\begin{theorem}[Expectation Binomial Distribution]
Suppose that  $X\sim Bin(n,p)$. Then $E[X]=n\cdot p.$
\end{theorem}

\begin{theorem}[Expectation Geometric Distribution]
Suppose that  $X\sim Geometric(p)$. Then $E[X]=1/p.$
\end{theorem}

\end{frame}

\begin{frame} %#19
\frametitle{Chebychev's Inequality}


\begin{theorem}
Let $X$ be a RV on a sample space $\Omega$ with probability mass function $p_{X}.$ If $r$ is a positive real number, then for every $\omega$
in $\Omega$ we have
$$
p_X(|X(\omega)-E(X)|\geq r)\leq V(X)/r^2.
$$
\end{theorem}

\begin{frame}
\frametitle{Täthetsfunktion (PDF-\textit{probability density function}) och sannolikheter}

\textcolor{blue}{Definition:} En reell-värd funktion $f:\mathbb{R}\to\mathbb{R}$ sådan att
\begin{enumerate}
    \item $f(x)\geq 0$ för alla reella tal $x\in \mathbb{R}$, och

    \item $\int_{-\infty}^{\infty}f(x)dx=1$
\end{enumerate} 
kallas för en \textcolor{blue}{täthetsfunktion} (probaility density function (PDF)).

%Det finns oändligt många sådana funktioner, men i praktiken så är
%det bara ett fåtal som är intressanta för tillämpningar!

\bigskip\pause
Om $X$ är en kontinuerlig stokastisk variabelså är \textcolor{blue}{$f_X$ täthetsfunktion till $X$ om sannolikheten}
att $X$ antar värde i intervall $[a,b]$ är
$$
P(a<X\leq b)=\int_{a}^{b}f_X(x)dx,
$$
för alla möjliga intervall $[a,b]$.

\bigskip\pause
Obs! $P(X=a)=0$ för alla värden på $a$ så för kontinuerliga s.v. är alltid
$$
P(X\leq a)=P(X<a).
$$

\end{frame}

\begin{frame}
Låt $X$ var den stokastisk variabelvars täthetsfunktion är
$$
f_X(x)=
\begin{cases}
\frac{3}{2}x^2,  \quad -1 \leq  x\leq 1,\\
0 \quad\quad\quad  \text{f.ö.}
\end{cases}
$$
Man gör ett experiment. Bestäm sannolikheten att utfallet blir att $X$ antar ett värde i intervallet $-0.25 < x < 0.75$.

\bigskip
Lösning:

\pause
\begin{figure}
\includegraphics[scale=0.045]{kvadratarea.jpeg} 
\end{figure}


\pause
\begin{equation*}
\begin{split}
P(-0.25< X<0.75)&=\int_{-0.25}^{0.75}\frac{3}{2}x^2dx=\frac{3}{2}\left[\frac{1}{3}x^3\right]_{-0.25}^{0.75}\\
&=\frac{1}{2}\left[x^3\right]_{-0.25}^{0.75}
=\frac{1}{2}[\frac{27}{64}-(-\frac{1}{64})]=\frac{1}{2}\cdot\frac{28}{64}=\frac{7}{32}(\approx 0.22).
\end{split}
\end{equation*}

$$

$$
\pause För diskreta stokastiska variabler som kan anta värdena $0,1,2,\dots k$

$$

$$
\pause Allmänt gäller att
$$
P(a<X\leq b)=F(b)-F(a).
